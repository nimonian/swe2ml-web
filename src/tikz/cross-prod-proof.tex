\begin{tikzpicture}[scale=1.5]
  \coordinate (O) at (0,0);
  \coordinate (A) at (1,3);
  \coordinate (B) at (4,2);

  \draw[vp-path, color=c-indigo-1, fill=c-indigo-1, fill opacity=0.1] (O) 
  -- (A)
  -- (0, 3) node[midway, above, opacity=1] {$v_1$}
  -- (O) node[midway, left, opacity=1] {$v_2$};
  
  \draw[vp-path, color=c-red-1, fill=c-red-1, fill opacity=0.1] (O)
  -- (B)
  -- (4, 0) node[midway, right, opacity=1] {$u_2$}

  \draw[vp-path, color=c-green-1, fill=c-green-1, fill opacity=0.1] (A)
  -- (4,3) node[midway, above, opacity=1] {$u_1 - v_1$}
  -- (B) node[midway, right, opacity=1] {$v_2 - u_2$};

  \draw[vp-vector] (O) -- (A) node[midway, above left] {$\vec{v}$};
  \draw[vp-vector] (O) -- (B) node[midway, below right] {$\vec{u}$};
  \draw[vp-vector, dashed] (B) -- (A) node[midway, sloped, below] {$\vec{v} - \vec{u}$};

  \path pic["$\theta$", vp-angle, angle radius=30pt] {angle = B--O--A};
